### I | 波動率燈號協議(v9.0 新增)

執行以下 Python 腳本(用 bash_tool),自動取得週/月線 BBand 並計算紅綠燈:

```python
import yfinance as yf
import numpy as np

ticker = "TICKER_PLACEHOLDER"  # 替換為實際標的

# 取得週線和月線數據
weekly = yf.download(ticker, period="24wk", interval="1wk")
monthly = yf.download(ticker, period="24mo", interval="1mo")

def compute_bband(prices, periods=20):
    closes = prices['Close'].iloc[-periods:].values
    sma = np.mean(closes)
    std = np.std(closes, ddof=0)
    return {
        'sma': sma,
        'upper_2': sma + 2 * std,
        'upper_3': sma + 3 * std,
        'lower_2': sma - 2 * std,
        'lower_3': sma - 3 * std,
    }

wk_bb = compute_bband(weekly)
mo_bb = compute_bband(monthly)
current_price = float(weekly['Close'].iloc[-1])

# 紅燈累加
red = 0
if current_price > wk_bb['upper_2']: red += 1
if current_price > wk_bb['upper_3']: red += 1
if current_price > mo_bb['upper_2']: red += 1
if current_price > mo_bb['upper_3']: red += 1

# 綠燈累加
green = 0
if current_price < wk_bb['lower_2']: green += 1
if current_price < wk_bb['lower_3']: green += 1
if current_price < mo_bb['lower_2']: green += 1
if current_price < mo_bb['lower_3']: green += 1

print(f"當前價格: {current_price:.2f}")
print(f"週線 BB: 2σ上軌 {wk_bb['upper_2']:.2f}, 3σ上軌 {wk_bb['upper_3']:.2f}")
print(f"月線 BB: 2σ上軌 {mo_bb['upper_2']:.2f}, 3σ上軌 {mo_bb['upper_3']:.2f}")
print(f"紅燈: {red}/4, 綠燈: {green}/4")
```
